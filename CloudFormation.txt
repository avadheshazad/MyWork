Description: Create an EC2 instance with a security group for SSH access And Attach an EBS volume to the EC2 instance. Add an Amazon S3 bucket.

Resources:
  InstanceSecurityGroup:
    Type: AWS::EC2::SecurityGroup
    Properties:
      GroupDescription: Enable SSH access
      SecurityGroupIngress:
        - IpProtocol: tcp
          FromPort: 22
          ToPort: 22
          CidrIp: 0.0.0.0/0

  MyInstance:
    Type: AWS::EC2::Instance
    Properties:
      ImageId: ami-0cfb394ad3c3ac699  
      InstanceType: t2.micro
      SecurityGroups:
        - !Ref InstanceSecurityGroup

  MyVolume:
    Type: AWS::EC2::Volume
    Properties:
      AvailabilityZone: !GetAtt MyInstance.AvailabilityZone
      Size: 6

  MyVolumeAttachment:
    Type: AWS::EC2::VolumeAttachment
    Properties:
      InstanceId: !Ref MyInstance
      VolumeId: !Ref MyVolume
      Device: /dev/sdf

  MyS3Bucket:
    Type: AWS::S3::Bucket
    Properties:
      BucketName: aaz-uni-bucket-name-4rw3dda34xxxaaz999
